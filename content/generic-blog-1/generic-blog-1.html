            org-sidenote.el --- Convert Org footnotes to simple tufte style sidenotes
            

            (
            require '
            ox-html
            )


            (
            defgroup 
            org-sidenote nil
  
            "Options for exporting Org files with sidenotes."
  
            :tag 
            "Org Sidenote"
  
            :group 'org-export
            )


            (
            defun 
            org-sidenote--get-footnote-definition 
            (footnote-reference info
            )
  
            "Extract the footnote definition for FOOTNOTE-REFERENCE using INFO."
  
            (
            let 
            (
            (definition 
            (org-export-get-footnote-definition footnote-reference info
            )
            )
            )
    
            (
            when definition
      
            ;; 
            Get the exported content and strip paragraph tags
                  
            (
            let 
            (
            (content 
            (org-trim 
            (org-export-data definition info
            )
            )
            )
            )
        
            ;; 
            Remove wrapping <p class="footpara"> tags if present
                    
            (replace-regexp-in-string
         
            "\\`<p class=\"footpara\">
            
              \\
            
            
              (
            
            
              \\
            
            
              (?:
            
            .
            
              \\
            
            
              |
            
            \n
            
              \\
            
            
              )
            
            *?
            
              \\
            
            
              )
            
            </p>\\'"
         
            "\\1"
         content
            )
            )
            )
            )
            )


            (
            defun 
            org-sidenote-footnote-reference 
            (footnote-reference _contents info
            )
  
            "Transcode a FOOTNOTE-REFERENCE element into a sidenote.
CONTENTS is nil.  INFO is a plist holding contextual information."
  
            (
            let* 
            (
            (number 
            (org-export-get-footnote-number footnote-reference info
            )
            )
         
            (definition 
            (org-sidenote--get-footnote-definition footnote-reference info
            )
            )
         
            (id 
            (format 
            "sn-%d" number
            )
            )
            )

    
            (
            if definition
        
            (format 
            "<label for=\"%s\" class=\"margin-toggle sidenote-number\"></label><input type=\"checkbox\" id=\"%s\" class=\"margin-toggle\"/><span class=\"sidenote\">%s</span>"
                id id definition
            )
      
            ;; 
            Fallback if no definition found
                  
            (format 
            "<sup>%d</sup>" number
            )
            )
            )
            )


            (
            defun 
            org-sidenote-fix-toc-html 
            (html
            )
  
            "Fix Org's malformed TOC HTML structure."
  
            (replace-regexp-in-string 
   
            "
            
              \\
            
            
              (
            
            <ul>[[:space:]\n]*<li>.*?</li>[[:space:]\n]*</ul>
            
              \\
            
            
              )
            
            [[:space:]\n]*</li>"
   
            "\\1"
   html t t
            )
            )




            (
            defun 
            org-sidenote-template 
            (contents info
            )
  
            "Return complete document string after HTML conversion.
CONTENTS is the transcoded contents string. INFO is a plist
holding export options."
  
            (
            let 
            (
            (html-template 
            (org-html-template contents info
            )
            )
            )
    
            ;; 
            Wrap title in header tags
                
            (
            setq html-template
          
            (replace-regexp-in-string
           
            "<div id=\"content\"[
            
              ^
            
            >]*>\\s-*<h1 class=\"title\">"
           
            "<div id=\"content\"><header><h1 class=\"title\">"
           html
            )
            )
    
            ;; 
            Wrap TOC in Nav and main tags
                
            (
            setq html-template
          
            (replace-regexp-in-string
           
            (concat 
            "<div id=\"table-of-contents\" role=\"doc-toc\">\n"
                   
            " *<h2>Table of Contents</h2>\n"
                   
            " *<div id=\"text-table-of-contents\" role=\"doc-toc\">\n"
                   
            " *
            
              \\
            
            
              (
            
            <ul>
            
              \\
            
            
              (?:
            
            .
            
              \\
            
            
              |
            
            \n
            
              \\
            
            
              )
            
            *?</ul>
            
              \\
            
            
              )
            
            \n"
                   
            " *</div>\n"
                   
            " *</div>"
            )
           
            (concat 
            "<nav id=\"table-of-contents\" role=\"doc-toc\">\n"
                   
            " *<input type=\"checkbox\" id=\"contents\">\n"
                   
            " *\\1\n"
                   
            " *</nav>\n"
                   
            " *<main>"
            )
           html t t
            )
            )
    
            ;; 
            Add closing main tag
            
    
            (
            setq html-template
      
            (replace-regexp-in-string 
            "<div id=\"postamble\" class=\"status\">" 
            "</main>\n <div id=\"postamble\" class=\"status\">" html
            )
            )
            )
            ) 


            ;; 
            Define the custom export backend
            
            (org-export-define-derived-backend 'sidenote 'html
  
            :translate-alist '
            (
            (footnote-reference . org-sidenote-footnote-reference
            )
                     
            (inner-template . org-sidenote-inner-template
            )
                     
            (template . org-sidenote-template
            )
            )
  
            :options-alist
  '
            (
            (
            :html-head-include-default-style nil 
            "html-style" nil
            )
    
            (
            :html-head-include-scripts nil 
            "html-scripts" nil
            )
            )
  
            :menu-entry
  '
            (?s 
            "Export with sidenotes"
       
            (
            (?h 
            "As HTML buffer" org-sidenote-export-to-html
            )
        
            (?H 
            "As HTML file" org-sidenote-export-to-html-file
            )
        
            (?o 
            "As HTML file and open"
            
            (
            lambda 
            (a s v b
            )
              
            (
            if a 
            (org-sidenote-export-to-html-file t s v b
            )
                
            (org-open-file 
            (org-sidenote-export-to-html-file nil s v b
            )
            )
            )
            )
            )
            )
            )
            )


            (
            defun 
            org-sidenote-export-to-html 
            (
            &optional async subtreep visible-only body-only ext-plist
            )
  
            "Export current buffer to HTML buffer with sidenotes."
  
            (
            interactive
            )
  
            (org-export-to-buffer 'sidenote 
            "*Org Sidenote HTML Export*"
    async subtreep visible-only body-only ext-plist 
            (
            lambda 
            () 
            (html-mode
            )
            )
            )
            )


            (
            defun 
            org-sidenote-export-to-html-file 
            (
            &optional async subtreep visible-only body-only ext-plist
            )
  
            "Export current buffer to HTML file with sidenotes."
  
            (
            interactive
            )
  
            (
            let 
            (
            (file 
            (org-export-output-file-name 
            ".html" subtreep
            )
            )
            )
    
            (org-export-to-file 'sidenote file
      async subtreep visible-only body-only ext-plist
            )
            )
            )

            provide '
            org-sidenote
